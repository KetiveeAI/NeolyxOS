Vector Layout

Vector Type Description
0-31 Exceptions CPU exceptions (fault/trap/abort)
32-47 IRQs Hardware interrupts (PIC remapped)
48-255 User Software/APIC interrupts

IDT Entry Structure

typedef struct {
    uint16_t offset_low;     // Handler address [15:0]
    uint16_t selector;       // Code segment selector
    uint8_t  ist;            // Interrupt Stack Table
    uint8_t  type_attr;      // Type and attributes
    uint16_t offset_mid;     // Handler address [31:16]
    uint32_t offset_high;    // Handler address [63:32]
    uint32_t reserved;
} __attribute__((packed)) idt_entry_t;

ISR Stub (Assembly)

; ISR stub saves all registers, calls C handler
isr_stub_N:
    push 0              ; Error code (if none)
    push N              ; Vector number
    jmp isr_common

isr_common:
    push rax, rbx, rcx, rdx, rsi, rdi, rbp, r8-r15
    mov rdi, rsp        ; Pointer to interrupt frame
    call interrupt_handler
    pop r15-r8, rbp, rdi, rsi, rdx, rcx, rbx, rax
    add rsp, 16         ; Pop error code + vector
    iretq
