Variables & Types

REOX uses let for immutable and let mut for mutable variables.

// Immutable (default)
let x: int = 42;
let name: string = "NeolyxOS";
let active: bool = true;
let pi: float = 3.14159;

// Mutable
let mut count: int = 0;
count = count + 1;

Type Inference

let x = 42;           // Inferred as int
let text = "hello";   // Inferred as string

Arrays

let numbers: [int] = [1, 2, 3, 4, 5];
let first = numbers[0];  // 1

Functions

fn add(a: int, b: int) -> int {
    return a + b;
}

// Expression body (short form)
fn multiply(a: int, b: int) -> int = a * b;

// Async functions
async fn fetch_data(url: string) -> string {
    await http.get(url)
}

Structs

struct User {
    id: int,
    name: string,
    email: string
}

let user = User {
    id: 1,
    name: "Swana",
    email: "swana@neolyx.os"
};

print(user.name);  // "Swana"

Kinds (Tagged Unions)

REOX uses kind instead of enum for tagged unions with associated data.

kind Color {
    Red,
    Green,
    Blue,
    Rgb(int, int, int)
}

kind Result {
    Ok(string),
    Error(string)
}

let color = Color::Rgb(255, 128, 0);

Control Flow

If/Else

if x > 10 {
    print("Large");
} else if x > 5 {
    print("Medium");
} else {
    print("Small");
}

While Loop

let mut i = 0;
while i < 10 {
    print(i);
    i = i + 1;
}

For Loop

for item in [1, 2, 3, 4, 5] {
    print(item);
}

Pattern Matching

match color {
    when Color::Red => print("Red!"),
    when Color::Rgb(r, g, b) => {
        print("RGB: " + r + ", " + g + ", " + b);
    },
    _ => print("Other color")
}

Actions (Closures)

// Named action
let greet = action(name: string) -> string {
    return "Hello, " + name + "!";
};

print(greet("NeolyxOS"));

// Short inline action
let double = action(x) { x * 2 };

// Use in higher-order functions
let doubled = numbers.map(action(x) { x * 2 });

Maybe (Optionals)

let user: maybe User = find_user(42);

if let Some(u) = user {
    print("Found: " + u.name);
} else {
    print("User not found");
}

// Optional chaining
let name = user?.name ?? "Unknown";
