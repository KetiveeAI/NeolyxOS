🧮 Calculator

A simple calculator with reactive state.

panel Calculator {
    title: "Calculator"
    size: (300, 400)
    
    @Bind display: string = "0"
    @Bind current: int = 0
    @Bind op: string = ""
    
    fn append(digit: string) {
        if display == "0" {
            display = digit;
        } else {
            display = display + digit;
        }
    }
    
    fn calculate() {
        let val = parse_int(display);
        match op {
            when "+" => display = to_string(current + val),
            when "-" => display = to_string(current - val),
            when "*" => display = to_string(current * val),
            when "/" => display = to_string(current / val),
            _ => {}
        }
    }
    
    vstack(gap: 8) {
        // Display
        text(display)
            .font_size(32)
            .align("right")
            .padding(16)
            .background(0x21262D)
            .corner_radius(8)
        
        // Number grid
        grid(cols: 4, gap: 8) {
            for n in ["7","8","9","/","4","5","6","*","1","2","3","-","C","0","=","+"] {
                button(n)
                    .padding(20)
                    .font_size(18)
                    .on_click(action {
                        match n {
                            when "C" => { display = "0"; }
                            when "=" => calculate()
                            when "+" | "-" | "*" | "/" => {
                                current = parse_int(display);
                                op = n;
                                display = "0";
                            }
                            _ => append(n)
                        }
                    })
            }
        }
    }
}

📝 Todo List

A todo list with add/remove functionality.

struct Todo {
    id: int,
    text: string,
    done: bool
}

panel TodoApp {
    title: "Todo List"
    size: (400, 500)
    
    @Bind todos: [Todo] = []
    @Bind input_text: string = ""
    @Bind next_id: int = 1
    
    fn add_todo() {
        if input_text != "" {
            todos = todos + [Todo { id: next_id, text: input_text, done: false }];
            next_id = next_id + 1;
            input_text = "";
        }
    }
    
    vstack(gap: 16) {
        text("📝 My Todos").font_size(24).font_weight("bold")
        
        // Input row
        hstack(gap: 8) {
            input()
                .placeholder("Add a todo...")
                .bind(input_text)
                .flex(1)
            button("Add").on_click(action { add_todo() })
        }
        
        divider()
        
        // Todo list
        scroll() {
            for todo in todos {
                hstack(gap: 12) {
                    checkbox(todo.done)
                        .on_change(action(v) { todo.done = v })
                    text(todo.text)
                        .strikethrough(todo.done)
                        .color(if todo.done { 0x8B949E } else { 0xC9D1D9 })
                        .flex(1)
                    button("🗑")
                        .on_click(action { 
                            todos = todos.filter(action(t) { t.id != todo.id })
                        })
                }.padding(8)
            }
        }
        
        text(len(todos) + " items").color(0x8B949E)
    }
}

🖥️ System Monitor

Display CPU and memory usage (NeolyxOS system app).

import system.monitor

panel SystemMonitor {
    title: "System Monitor"
    size: (500, 400)
    
    @Bind cpu_usage: float = 0.0
    @Bind mem_usage: float = 0.0
    @Bind disk_usage: float = 0.0
    
    fn update() {
        cpu_usage = monitor.get_cpu();
        mem_usage = monitor.get_memory();
        disk_usage = monitor.get_disk();
    }
    
    // Update every second
    effect {
        timer.interval(1000, action { update() });
    }
    
    vstack(gap: 20) {
        text("🖥️ NeolyxOS System Monitor")
            .font_size(20)
            .font_weight("bold")
        
        // CPU Usage
        vstack(gap: 4) {
            hstack {
                text("CPU").font_weight("bold")
                spacer()
                text(format("%.1f%%", cpu_usage))
            }
            progress(cpu_usage / 100)
                .color(0x58A6FF)
                .height(8)
        }
        
        // Memory Usage
        vstack(gap: 4) {
            hstack {
                text("Memory").font_weight("bold")
                spacer()
                text(format("%.1f%%", mem_usage))
            }
            progress(mem_usage / 100)
                .color(0x3FB950)
                .height(8)
        }
        
        // Disk Usage
        vstack(gap: 4) {
            hstack {
                text("Disk").font_weight("bold")
                spacer()
                text(format("%.1f%%", disk_usage))
            }
            progress(disk_usage / 100)
                .color(0xD29922)
                .height(8)
        }
    }.padding(24)
}

📁 File Browser

Browse files with tree navigation.

import system.fs

kind FileEntry {
    File(string),
    Directory(string)
}

panel FileBrowser {
    title: "Files"
    size: (600, 500)
    
    @Bind current_path: string = "/"
    @Bind entries: [FileEntry] = []
    
    fn load_dir(path: string) {
        current_path = path;
        entries = fs.list_dir(path);
    }
    
    fn icon_for(entry: FileEntry) -> string {
        match entry {
            when FileEntry::Directory(_) => "📁",
            when FileEntry::File(name) => {
                if name.ends_with(".rx") { "📄" }
                else if name.ends_with(".png") { "🖼️" }
                else { "📝" }
            }
        }
    }
    
    vstack(gap: 0) {
        // Path bar
        hstack(gap: 8) {
            button("⬆️ Up").on_click(action {
                load_dir(fs.parent(current_path))
            })
            text(current_path).flex(1).font_family("monospace")
        }.padding(12).background(0x21262D)
        
        divider()
        
        // File list
        scroll() {
            for entry in entries {
                hstack(gap: 8) {
                    text(icon_for(entry))
                    text(match entry {
                        when FileEntry::File(n) | FileEntry::Directory(n) => n
                    }).flex(1)
                }
                .padding(12)
                .on_click(action {
                    match entry {
                        when FileEntry::Directory(path) => load_dir(path),
                        _ => {}
                    }
                })
                .on_hover(action { /* highlight */ })
            }
        }.flex(1)
    }
}
