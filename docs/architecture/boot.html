Boot Flow

UEFI Firmware
    │
    ▼
BOOTX64.EFI (boot/main.c)
    ├── Initialize GOP (graphics)
    ├── Load kernel.bin from ESP
    ├── Validate ELF header
    ├── Get memory map
    ├── Find ACPI RSDP
    ├── Prepare NeolyxBootInfo
    ├── ExitBootServices()
    │
    ▼
kernel_main(NeolyxBootInfo*)
    ├── Validate magic (0x4E454F58)
    ├── Init serial (COM1)
    ├── Init framebuffer
    ├── Init IDT/GDT
    ├── Init PMM
    ├── Init drivers
    └── Start scheduler

Boot Info Structure

#define NEOLYX_BOOT_MAGIC 0x4E454F58  // "NEOX"

typedef struct {
    uint32_t magic;                 // Must be NEOLYX_BOOT_MAGIC
    uint32_t version;               // Boot protocol version
    
    // Memory map from UEFI
    uint64_t memory_map_addr;
    uint64_t memory_map_size;
    uint64_t memory_map_desc_size;
    uint32_t memory_map_desc_version;
    
    // Framebuffer info
    uint64_t framebuffer_addr;
    uint32_t framebuffer_width;
    uint32_t framebuffer_height;
    uint32_t framebuffer_pitch;
    uint32_t framebuffer_bpp;
    
    // Kernel info
    uint64_t kernel_physical_base;
    uint64_t kernel_size;
    
    // ACPI
    uint64_t acpi_rsdp;
} NeolyxBootInfo;

Kernel Entry

void kernel_main(NeolyxBootInfo *boot_info) {
    // 1. Validate boot info
    if (boot_info->magic != NEOLYX_BOOT_MAGIC) {
        // Invalid boot - halt
        while(1) __asm__("hlt");
    }
    
    // 2. Initialize serial console (debug)
    serial_init();
    serial_puts("NeolyxOS booting...\n");
    
    // 3. Initialize framebuffer
    fb_init(boot_info->framebuffer_addr,
            boot_info->framebuffer_width,
            boot_info->framebuffer_height,
            boot_info->framebuffer_pitch);
    
    // 4. Initialize memory from UEFI map
    pmm_init(boot_info->memory_map_addr,
             boot_info->memory_map_size,
             boot_info->memory_map_desc_size);
    
    // 5. Initialize interrupts
    gdt_init();
    idt_init();
    
    // 6. Initialize drivers
    drivers_init();
    
    // 7. Start scheduler
    scheduler_init();
}

UEFI Memory Map

Type Usage
EfiConventionalMemory Free, usable RAM
EfiLoaderCode/Data Bootloader (reclaimable)
EfiBootServicesCode/Data UEFI services (reclaimable)
EfiRuntimeServicesCode/Data UEFI runtime (reserved)
EfiACPIReclaimMemory ACPI tables (reclaimable)
EfiReservedMemoryType Reserved by firmware
