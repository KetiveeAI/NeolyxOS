Required drivers, APIs, and capabilities for building a file manager.

File Operations API

// List directory
vfs_dir_t* dir = vfs_opendir("/Users/alice/Documents");
vfs_dirent_t entry;
while (vfs_readdir(dir, &entry) == 0) {
    printf("%s  %s  %lu bytes\n", 
        entry.type == VFS_DIR ? "[DIR]" : "[FILE]",
        entry.name,
        entry.size);
}
vfs_closedir(dir);

// Copy file
int src = vfs_open("/path/source.txt", VFS_READ);
int dst = vfs_open("/path/dest.txt", VFS_WRITE | VFS_CREATE);
char buf[4096];
int n;
while ((n = vfs_read(src, buf, sizeof(buf))) > 0) {
    vfs_write(dst, buf, n);
}
vfs_close(src);
vfs_close(dst);

// Get file info
vfs_stat_t info;
vfs_stat("/path/file.txt", &info);
printf("Size: %lu, Modified: %lu\n", info.size, info.mtime);

File Manager Capabilities

// Info.plist for File Manager app
{
    "NXBundleIdentifier": "com.neolyx.finder",
    "NXBundleName": "Finder",
    "NXCapabilities": [
        "filesystem.full",     // Full disk access
        "filesystem.admin",    // Admin operations
        "network.share",       // Network file sharing
        "removable.mount"      // Mount external drives
    ]
}

  • ✓ VFS layer for filesystem abstraction
  • ✓ NXFS read/write support
  • ✓ Directory listing (readdir)
  • ✓ File copy/move/delete
  • ✓ SIP enforcement (can't modify /System)

  • □ Search indexing

Network File Access

// Mount network share
vfs_mount("nxshare://server/share", "/Volumes/Share", "nxnet", 0);

// Access like local files
vfs_open("/Volumes/Share/file.txt", VFS_READ);

// Unmount
vfs_unmount("/Volumes/Share");
