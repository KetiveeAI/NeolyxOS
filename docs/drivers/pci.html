PCI Device Structure

typedef struct {
    uint8_t  bus, device, function;
    uint16_t vendor_id;
    uint16_t device_id;
    uint8_t  class_code;
    uint8_t  subclass;
    uint8_t  prog_if;
    uint8_t  revision;
    uint8_t  interrupt_line;
    uint8_t  interrupt_pin;
    
    // Base Address Registers
    uint8_t  bar_count;
    pci_bar_t bars[6];
} pci_device_t;

API

// Initialize PCI bus
void pci_init(void);

// Find devices
pci_device_t *pci_find_device(uint16_t vendor, uint16_t device);
pci_device_t *pci_find_class(uint8_t class, uint8_t subclass);
int pci_get_device_count(void);

// Config space access
uint8_t  pci_read_config8(uint8_t bus, uint8_t dev, uint8_t func, uint8_t off);
uint16_t pci_read_config16(uint8_t bus, uint8_t dev, uint8_t func, uint8_t off);
uint32_t pci_read_config32(uint8_t bus, uint8_t dev, uint8_t func, uint8_t off);

void pci_write_config8(uint8_t bus, uint8_t dev, uint8_t func, uint8_t off, uint8_t val);
void pci_write_config16(uint8_t bus, uint8_t dev, uint8_t func, uint8_t off, uint16_t val);
void pci_write_config32(uint8_t bus, uint8_t dev, uint8_t func, uint8_t off, uint32_t val);

// Enable device
void pci_enable_bus_master(pci_device_t *dev);
void pci_enable_memory(pci_device_t *dev);

Config Space Registers

Offset Size Register
0x00 2 Vendor ID
0x02 2 Device ID
0x04 2 Command
0x06 2 Status
0x08 1 Revision
0x09 1 Prog IF
0x0A 1 Subclass
0x0B 1 Class Code
0x10-0x24 4 each BAR 0-5
0x3C 1 Interrupt Line
0x3D 1 Interrupt Pin

Usage Example

// Find NVMe controller
pci_device_t *nvme = pci_find_class(0x01, 0x08);
if (nvme) {
    pci_enable_bus_master(nvme);
    pci_enable_memory(nvme);
    
    // Get BAR0 (MMIO registers)
    uint64_t bar0 = nvme->bars[0].base;
    volatile uint32_t *regs = (volatile uint32_t*)bar0;
}
