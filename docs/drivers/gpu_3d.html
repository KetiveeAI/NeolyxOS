Command Buffer

NXRender uses a ring buffer to queue GPU commands. This allows the CPU to continue preparing the next frame while the GPU executes the current one.

// Command buffer structure
typedef struct {
    nx_gpu_cmd_t commands[4096]; // 4096 commands
    volatile uint32_t write_index; // CPU writes here
    volatile uint32_t read_index;  // GPU reads here
    volatile uint32_t fence_value; // Synchronization
} nx_gpu_cmd_buffer_t;

API Reference

Basic Usage

// Initialize GPU driver
nx_gpu_driver_t* gpu = nx_gpu_driver_init();

// Enable VSync (prevents tearing)
nx_gpu_set_vsync(gpu, true);

// Main render loop
while (running) {
    // Begin frame
    nx_gpu_begin_frame(gpu);

    // Queue draw commands
    nx_gpu_cmd_clear(gpu, NX_COLOR_BLACK);
    nx_gpu_cmd_rect(gpu, (nx_rect_t){100, 100, 200, 150}, NX_COLOR_WHITE, 8);
    nx_gpu_cmd_circle(gpu, (nx_point_t){400, 300}, 50, NX_COLOR_BLUE);

    // End frame and present
    nx_gpu_end_frame(gpu);
    nx_gpu_present(gpu); // Waits for VSync
}

// Cleanup
nx_gpu_driver_shutdown(gpu);

Draw Commands

Function Description
nx_gpu_cmd_clear Clear framebuffer
nx_gpu_cmd_rect Draw rectangle (rounded)
nx_gpu_cmd_circle Draw filled circle
nx_gpu_cmd_line Draw line
nx_gpu_cmd_blit Draw texture
nx_gpu_cmd_mesh Draw 3D mesh

Texture Management

// Create texture
nx_gpu_texture_t tex = nx_gpu_create_texture(gpu, 256, 256, NX_TEX_RGBA8);

// Upload pixel data
uint32_t* pixels = malloc(256 * 256 * 4);
// ... fill pixels ...
nx_gpu_upload_texture(gpu, tex, pixels);
free(pixels);

// Use texture
nx_gpu_cmd_blit(gpu, tex, (nx_rect_t){0, 0, 256, 256});

// Cleanup
nx_gpu_destroy_texture(gpu, tex);

Supported Formats

Format Bytes Description

NX_TEX_BGRA8 4 Windows-style byte order
NX_TEX_RGB8 3 24-bit color, no alpha
NX_TEX_R8 1 Grayscale
NX_TEX_DEPTH24 3 Depth buffer

Advanced

Buffer Management

// Create vertex buffer
nx_gpu_buffer_t vbo = nx_gpu_create_buffer(gpu, sizeof(nx_vertex_t) * 1000, 0);

// Map for CPU access
nx_vertex_t* vertices = nx_gpu_map_buffer(gpu, vbo);
vertices[0] = (nx_vertex_t){ ... };
nx_gpu_unmap_buffer(gpu, vbo);

// Or upload directly
nx_gpu_upload_buffer(gpu, vbo, data, size);

Synchronization

Kernel Integration

The driver uses standard POSIX-like calls to communicate with the kernel implementation:

gpu->driver_fd = open("/dev/nxgpu", O_RDWR);
gpu->cmd_buffer = mmap(NULL, cmd_size, ..., gpu->driver_fd, CMD_BUFFER_OFFSET);
gpu->vram_base = mmap(NULL, vram_size, ..., gpu->driver_fd, VRAM_OFFSET);
