💾 Block Layer Complete

Generic block device abstraction for storage drivers.

Purpose

The block layer provides a unified interface for block devices. Filesystems and VFS use this abstraction - they don't care if the backend is NVMe, AHCI, or USB.

Files

File Purpose
kernel/src/drivers/block/block.h Block device interface
kernel/src/drivers/block/block.c Device registry + I/O
kernel/src/drivers/block/nvme_block.c NVMe backend

Block Device Structure

typedef struct block_device {
    char          name[32];       // "nvme0n1", "sda"
    block_type_t  type;           // NVME, AHCI, ATA, USB
    uint32_t      flags;          // REMOVABLE, READONLY, PRESENT
    
    uint64_t      total_blocks;
    uint32_t      block_size;     // 512 or 4096
    
    block_ops_t   *ops;           // Read/write/flush
    void          *private_data;  // Backend context
    
    int           index;
} block_device_t;

Block Operations

typedef struct {
    int (*read)(block_device_t *dev, uint64_t lba, uint32_t count, void *buffer);
    int (*write)(block_device_t *dev, uint64_t lba, uint32_t count, const void *buffer);
    int (*flush)(block_device_t *dev);
    uint64_t (*get_capacity)(block_device_t *dev);
} block_ops_t;

API

Device Registration

// Initialize block subsystem
void block_init(void);

// Register a block device
int block_register(block_device_t *dev);
// Returns: device index

// Unregister
int block_unregister(int index);

Device Lookup

// Get device by index
block_device_t *block_get(int index);

// Get device by name
block_device_t *block_find(const char *name);

// Get device count
int block_count(void);

Block I/O

// Read blocks
int block_read(block_device_t *dev, uint64_t lba, uint32_t count, void *buffer);

// Write blocks
int block_write(block_device_t *dev, uint64_t lba, uint32_t count, const void *buffer);

// Flush buffers
int block_flush(block_device_t *dev);

// Get capacity in bytes
uint64_t block_capacity(block_device_t *dev);

Example Usage

// Initialize and register NVMe devices
block_init();
nvme_block_init();  // Registers NVMe namespaces

// Find device by name
block_device_t *disk = block_find("nvme0n1");

// Read first 8 sectors
uint8_t buffer[4096];
block_read(disk, 0, 8, buffer);

// Get capacity
uint64_t size = block_capacity(disk);
// size = total bytes on disk

How to Add New Backend

  1. Create kernel/src/drivers/block/mybackend_block.c
  2. Implement block_ops_t operations
  3. Allocate block_device_t and set fields
  4. Call block_register()
  5. Add to Makefile
