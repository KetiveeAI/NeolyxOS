Settings App Driver API

Unified driver access for the Settings application
Wraps kernel drivers for safe userspace access

Available Modules

Module Functions
Display Resolution, brightness, HDR, DDC/CI
Audio Volume, devices, effects
Network WiFi, Bluetooth, Ethernet
Power Battery, performance modes
Touch Gestures, trackpad, stylus
Laptop Backlight, keyboard backlight

Display Functions

#include "settings_drivers.h"

// Basic display
display_get_count();
display_get_info(0, &info);
display_set_mode(0, 1920, 1080, 144);
display_set_brightness(0, 80);

// DDC/CI (external monitors)
display_ddc_supported(0);
display_ddc_set_brightness(0, 70);
display_ddc_set_contrast(0, 50);

// Night Light
display_night_light_enable(0, 3400);  // Kelvin
display_night_light_disable(0);

// Gaming Mode
display_gaming_mode_enable(0);
display_gaming_mode_disable(0);
display_register_game("game.exe");

Laptop Brightness

// Check if laptop
if (laptop_is_laptop()) {
    laptop_set_brightness(80);
    laptop_set_brightness_smooth(50, 300);  // 300ms
}

// Auto-brightness
if (laptop_ambient_available()) {
    laptop_auto_brightness_enable(true);
}

// Keyboard backlight
if (laptop_keyboard_backlight_available()) {
    laptop_keyboard_backlight_set(50);
}

Touch Gestures

// Check capabilities
touch_available();
touch_is_touchscreen();
touch_is_touchpad();

// Settings
touch_set_natural_scroll(true);
touch_set_scroll_speed(5);
touch_set_palm_rejection(true);

// 3-finger actions (0=up, 1=down, 2=left, 3=right)
touch_set_three_finger_action(0, NXACTION_VOLUME_UP);

// 4-finger actions
touch_set_four_finger_action(0, NXACTION_MISSION_CONTROL);

// Edge gestures (0=left, 1=right, 2=top, 3=bottom)
touch_set_edge_gesture(3, NXACTION_APP_DRAWER);

// App switcher (Android-style)
touch_enable_app_switcher(true);

Audio Functions

audio_get_output_devices(devices, 16);
audio_set_master_volume(80);
audio_set_mute(false);
audio_get_settings(&settings);

Network Functions

// WiFi
settings_wifi_scan(networks, 32);
settings_wifi_connect("SSID", "password");
settings_wifi_disconnect();

// Network settings
network_get_settings(&settings);
network_set_airplane_mode(true);

Power Functions

power_get_info(&info);  // Battery level, charging
power_set_mode(POWER_MODE_BALANCED);
