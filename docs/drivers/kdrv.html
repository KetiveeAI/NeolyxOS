Driver Details

🖥️ nxgfx_kdrv - GPU/Framebuffer

Unified GPU interface supporting Intel, AMD, NVIDIA, and QEMU VGA.

// API
nxgfx_kdrv_init();
nxgfx_kdrv_set_mode(1920, 1080, 32);
nxgfx_kdrv_put_pixel(x, y, color);
nxgfx_kdrv_fill_rect(x, y, w, h, color);

🔊 nxaudio_kdrv - Audio

Supports Intel HD Audio (HDA) and AC97 codecs.

// API
nxaudio_kdrv_init();
int stream = nxaudio_kdrv_stream_open(48000, 2, 16);
nxaudio_kdrv_stream_start(stream);
nxaudio_kdrv_stream_write(stream, data, len);

🌐 nxnet_kdrv - Network

Supports E1000, VirtIO-Net, and RTL8169 NICs.

// API
nxnet_kdrv_init();
nxnet_kdrv_link_status(0);  // Check link
nxnet_kdrv_send(0, packet, len);
nxnet_kdrv_receive(0, buffer, max_len);

💾 nxstor_kdrv - Storage

Unified storage access for NVMe and ATA/AHCI devices.

// API
nxstor_kdrv_init();
nxstor_kdrv_read(device, lba, count, buffer);
nxstor_kdrv_write(device, lba, count, data);
nxstor_kdrv_stats(device, &stats);

🔌 nxusb_kdrv - USB

USB device enumeration and class detection.

// API
nxusb_kdrv_init();
int count = nxusb_kdrv_device_count();
nxusb_kdrv_device_info(0, &info);  // VID, PID, class

📊 nxpci_kdrv - PCI

PCI device enumeration and config space access.

// API
nxpci_kdrv_init();
nxpci_kdrv_enumerate();
uint32_t val = nxpci_kdrv_read_config(bus, dev, fn, reg);
nxpci_kdrv_write_config(bus, dev, fn, reg, val);

📋 nxsysinfo_kdrv - System Information

CPU info, thermal sensors, battery status, lid state.

// API
nxsysinfo_kdrv_init();
nxsysinfo_kdrv_cpu_info(&cpu);    // Vendor, cores, freq
int count = nxsysinfo_kdrv_thermal_count();
nxsysinfo_kdrv_thermal_info(0, &temp);  // CPU temp
nxsysinfo_kdrv_battery_info(&batt);

Features: CPUID parsing, enumeration-based thermals, cached static info

🖥️ nxdisplay_kdrv - Display Management

Multi-monitor detection, EDID parsing, atomic mode commits.

// API
nxdisplay_kdrv_init();
nxdisplay_kdrv_info(0, &display);  // EDID, resolution

// Atomic commit
nxdisplay_kdrv_set_mode(0, &mode);
nxdisplay_kdrv_set_position(0, 1920, 0);
nxdisplay_kdrv_set_color(0, NXCOLORSPACE_P3, 22);
nxdisplay_kdrv_commit(0);  // Apply all at once

Features: HDMI/DP/VGA/eDP, VRR, HDR, color profiles

✏️ nxpen_kdrv - Pen/Stylus Input

Digitizer support with pressure, tilt, and rotation.

// API
nxpen_kdrv_init();
nxpen_kdrv_digitizer_info(0, &digitizer);
nxpen_kdrv_pen_info(0, &pen);

// Event polling or callback
nxpen_kdrv_poll(&event);  // x, y, pressure, tilt
nxpen_kdrv_register_callback(my_handler, data);

Protocols: Wacom, Microsoft Pen (MPP), USI, N-Trig

📊 nxtask_kdrv - Task Manager

Process monitoring, CPU usage, priority control.

// API
nxtask_kdrv_init();

// System info
nxtask_kdrv_sysinfo(&sys);  // Load avg, CPU %, mem

// Process info
int count = nxtask_kdrv_count();
nxtask_kdrv_list(pids, max);
nxtask_kdrv_info(pid, &info);  // CPU %, memory, state

// Control
nxtask_kdrv_set_priority(pid, NXTASK_PRIO_HIGH);
nxtask_kdrv_set_nice(pid, -10);
nxtask_kdrv_suspend(pid);
nxtask_kdrv_kill(pid, SIGTERM);

Priority Levels: Idle, Low, Below Normal, Normal, Above Normal, High, Realtime

🖼️ nxrender_kdrv - GUI Rendering

Kernel-side interface for NXRender user-space GUI framework. Processes command buffers.

// API
nxrender_kdrv_init();

// Command buffer submission
nxrender_cmd_buffer_t* buf = nxrender_kdrv_get_cmd_buffer();
// ... populate buf->commands ...
nxrender_kdrv_submit();
nxrender_kdrv_present(0);

// Textures
int tex = nxrender_kdrv_create_texture(256, 256, NXRENDER_TEX_RGBA8);
nxrender_kdrv_upload_texture(tex, pixels, size);

// VSync
nxrender_kdrv_set_vsync(NXRENDER_VSYNC_ON);
nxrender_kdrv_wait_vblank();

Features: Command buffer, primitives (rect, circle, line), scissor, textures, VSync

Driver API Pattern

All drivers follow a consistent API pattern:

// Lifecycle
int  nx<name>_kdrv_init(void);
void nx<name>_kdrv_shutdown(void);

// Enumeration
int  nx<name>_kdrv_count(void);
int  nx<name>_kdrv_info(int idx, <type>_info_t *info);

// Debug
void nx<name>_kdrv_debug(void);

Adding to Makefile

# Object list
obj-y += nx<name>_kdrv.o

# Compile rule
nx<name>_kdrv.o: src/drivers/<subsystem>/nx<name>_kdrv.c include/drivers/nx<name>_kdrv.h
    $(CC) $(CFLAGS) -c $< -o $@
