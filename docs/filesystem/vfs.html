📁 Virtual File System Complete

Unified filesystem interface supporting multiple backends (NXFS, FAT32, NTFS).

Files

File Purpose
kernel/src/fs/vfs.c VFS core implementation
kernel/include/fs/vfs.h VFS structures and API
kernel/src/fs/nxfs.c NXFS native filesystem
kernel/src/drivers/block/block.c Block device layer

VFS Node Structure

typedef struct vfs_node {
    char name[256];
    uint32_t flags;         // VFS_FILE, VFS_DIRECTORY, VFS_MOUNT
    uint32_t inode;
    uint64_t size;
    
    // Operations
    vfs_ops_t *ops;
    
    // Links
    struct vfs_node *parent;
    struct vfs_node *children;
    struct vfs_node *next;
    
    // Backend-specific data
    void *private_data;
} vfs_node_t;

File Operations

typedef struct vfs_ops {
    int (*open)(vfs_node_t *node, uint32_t flags);
    int (*close)(vfs_node_t *node);
    int (*read)(vfs_node_t *node, void *buffer, size_t size, size_t offset);
    int (*write)(vfs_node_t *node, const void *buffer, size_t size, size_t offset);
    vfs_node_t *(*readdir)(vfs_node_t *node, uint32_t index);
    vfs_node_t *(*finddir)(vfs_node_t *node, const char *name);
    int (*create)(vfs_node_t *parent, const char *name, uint32_t flags);
    int (*unlink)(vfs_node_t *node);
} vfs_ops_t;

API

Path Operations

// Open file by path
vfs_node_t *vfs_open(const char *path, uint32_t flags);

// Close file
int vfs_close(vfs_node_t *node);

// Read from file
int vfs_read(vfs_node_t *node, void *buffer, size_t size, size_t offset);

// Write to file
int vfs_write(vfs_node_t *node, const void *buffer, size_t size, size_t offset);

Mount Operations

// Mount filesystem at path
int vfs_mount(const char *path, vfs_node_t *root, const char *type);

// Unmount filesystem
int vfs_unmount(const char *path);

// Get mount point count
int vfs_get_mount_count(void);

Directory Operations

// List directory contents
vfs_node_t *vfs_readdir(vfs_node_t *dir, uint32_t index);

// Find entry in directory
vfs_node_t *vfs_finddir(vfs_node_t *dir, const char *name);

// Create file/directory
int vfs_create(const char *path, uint32_t flags);

Supported Filesystems

Filesystem Read Write Notes
NXFS (Native) ✅ ✅ Extents, encryption, checksums
FAT32 ✅ 📋 LFN support
exFAT ✅ 📋 Large files
NTFS ✅ ❌ Read-only

Example Usage

// Mount root filesystem
vfs_mount("/", nxfs_root, "nxfs");

// Open a file
vfs_node_t *file = vfs_open("/home/user/test.txt", VFS_READ);

// Read contents
char buffer[1024];
int bytes = vfs_read(file, buffer, sizeof(buffer), 0);

// Close
vfs_close(file);

How to Add New Filesystem

  1. Create kernel/src/fs/myfs.c
  2. Implement vfs_ops_t operations
  3. Create mount function myfs_mount()
  4. Add to Makefile
  5. Register in VFS during boot
